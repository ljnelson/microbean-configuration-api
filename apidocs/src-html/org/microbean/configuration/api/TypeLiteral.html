017package org.microbean.configuration.api;
018
019import java.io.Serializable;
020
021import java.lang.reflect.ParameterizedType;
022import java.lang.reflect.Type;
023
024import java.util.Objects;
025
026/**
027 * A class that can represent generic types.
028 *
029 * @author <a href="http://about.me/lairdnelson"
030 * target="_parent">Laird Nelson</a>
031 *
032 * @param <T> the type being represented
033 *
034 * @see Type
035 */
036public abstract class TypeLiteral<T> implements Serializable {
037
038
039  /*
040   * Static fields.
041   */
042
043  
044  /**
045   * The version of this class for {@linkplain Serializable
046   * serialization purposes}.
047   */
048  private static final long serialVersionUID = 1L;
049
050
051  /*
052   * Instance fields.
053   */
054
055  
056  /**
057   * Within any given class hierarchy rooted with this class, the
058   * first class beneath the root whose immediate subclasses will be
059   * inspected for type information.
060   *
061   * <p>This field will never be {@code null}.</p>
062   *
063   * @see #TypeLiteral(Class)
064   */
065  private final Class<?> classHierarchyParent;
066
067  /**
068   * The actual {@link Type} housed by this {@link TypeLiteral}.
069   *
070   * <p>This field will never be {@code null}.</p>
071   */
072  private transient Type type;
073
074
075  /*
076   * Constructors.
077   */
078
079
080  /**
081   * Creates a new {@link TypeLiteral}.
082   */
083  protected TypeLiteral() {
084    this(TypeLiteral.class);
085  }
086
087  /**
088   * Creates a new {@link TypeLiteral} with a deeper class hierarchy.
089   *
090   * <p>This specialized constructor is for use only by {@code
091   * abstract} subclasses of {@link TypeLiteral} that are extending
092   * and looking to use the subsequently extended type inspection
093   * capabilities of this class.</p>
094   *
095   * @param <T> a type representing a subclass of {@link TypeLiteral}
096   *
097   * @param classHierarchyParent the first class beneath this class in
098   * the subclass hierarchy whose immediate subclasses will be
099   * inspected for parameterized type information; must not be {@code
100   * null}
101   *
102   * @exception NullPointerException if {@code classHierarchyParent}
103   * is {@code null}
104   */
105  protected <T extends TypeLiteral<?>> TypeLiteral(final Class<T> classHierarchyParent) {
106    super();
107    Objects.requireNonNull(classHierarchyParent);
108    this.classHierarchyParent = classHierarchyParent;
109  }
110
111  /**
112   * Returns the {@link Type} represented by this {@link TypeLiteral}.
113   *
114   * <p>This method never returns {@code null}.</p>
115   *
116   * @return the {@link Type} represented by this {@link TypeLiteral};
117   * never {@code null}
118   *
119   * @exception IllegalStateException if somehow this {@link
120   * TypeLiteral} was instantiated without a type parameter
121   */
122  public final Type getType() {
123    if (this.type == null) {
124      final Class<?> topmostTypeLiteralSubclass = getFirstSubclass(this.getClass(), this.classHierarchyParent);
125      assert topmostTypeLiteralSubclass != null;
126      this.type = getSoleTypeParameter(topmostTypeLiteralSubclass);
127      if (this.type == null) {
128        throw new IllegalStateException("No type parameter specified");
129      }
130    }
131    return this.type;
132  }
133
134  /**
135   * Returns a hashcode for this {@link TypeLiteral} based off its
136   * {@link #getType() Type}'s hashcode.
137   *
138   * @return a hashcode for this {@link TypeLiteral}
139   *
140   * @see #equals(Object)
141   */
142  @Override
143  public int hashCode() {
144    final Object type = this.getType();
145    return type == null ? 0 : type.hashCode();
146  }
147
148  /**
149   * Returns {@code true} if the supplied {@link Object} is equal to
150   * this {@link TypeLiteral}.
151   *
152   * <p>Two {@link TypeLiteral} instances are equal if their {@link
153   * #getType()} methods' return values {@linkplain
154   * Object#equals(Object) are equal}.</p>
155   *
156   * @param other the {@link Object} to test; may be {@code null}
157   *
158   * @return {@code true} if the supplied {@link Object} is equal to
159   * this {@link TypeLiteral}; {@code false} otherwise
160   *
161   * @see #hashCode()
162   */
163  @Override
164  public boolean equals(final Object other) {
165    if (other == this) {
166      return true;
167    } else if (other instanceof TypeLiteral) {
168      final TypeLiteral<?> her = (TypeLiteral<?>)other;
169      final Object type = this.getType();
170      if (type == null) {
171        if (her.getType() != null) {
172          return false;
173        }
174      } else if (!type.equals(her.getType())) {
175        return false;
176      }
177      return true;
178    } else {
179      return false;
180    }
181  }
182
183  /**
184   * Returns a non-{@code null} {@link String} representation of this
185   * {@link TypeLiteral}.
186   *
187   * <p>This method never returns {@code null}.</p>
188   *
189   * <p>Overrides of this method must not return {@code null}.</p>
190   *
191   * @return a non-{@code null} {@link String} representation of this
192   * {@link TypeLiteral}
193   */
194  @Override
195  public String toString() {
196    return String.valueOf(this.getType());
197  }
198
199  
200  /*
201   * Static methods.
202   */
203  
204
205  private static final Class<?> getFirstSubclass(final Class<?> c, final Class<?> classHierarchyParentType) {
206    final Class<?> returnValue;
207    if (c == null || c.equals(Object.class)) {
208      returnValue = null;
209    } else {
210      final Class<?> superclass = c.getSuperclass();
211      assert superclass != null;
212      if (superclass.equals(classHierarchyParentType)) {
213        returnValue = c;
214      } else if (superclass.equals(Object.class)) {
215        returnValue = null;
216      } else {
217        returnValue = getFirstSubclass(superclass, classHierarchyParentType); // recursive call
218      }
219    }
220    return returnValue;
221  }
222
223  private static final Type getSoleTypeParameter(final Class<?> c) {
224    Type returnValue = null;
225    if (c != null) {
226      final Type genericSuperclass = c.getGenericSuperclass();
227      if (genericSuperclass instanceof ParameterizedType) {
228        final ParameterizedType parameterizedType = (ParameterizedType)genericSuperclass;
229        final Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
230        if (actualTypeArguments != null && actualTypeArguments.length == 1) {
231          returnValue = actualTypeArguments[0];
232        }
233      }
234    }
235    return returnValue;
236  }
237
238}
